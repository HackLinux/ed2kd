CMAKE_MINIMUM_REQUIRED(VERSION 2.8)
PROJECT(ed2kd C)

INCLUDE(CMakeDependentOption)

CMAKE_DEPENDENT_OPTION(USE_SQLITE3 "Use SQLite3 database" OFF
    "NOT USE_MYSQL" OFF)
CMAKE_DEPENDENT_OPTION(USE_MYSQL "Use MySQL database" OFF
    "NOT USE_SQLITE3" OFF)

SET(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH}
    ${CMAKE_SOURCE_DIR}/cmake/modules)

FIND_PACKAGE(Libevent REQUIRED)
FIND_PACKAGE(Libconfig REQUIRED)
FIND_PACKAGE(ZLIB REQUIRED)
FIND_PACKAGE(Libatomic_ops REQUIRED)

SET(INCLUDES ${LIBEVENT_INCLUDE_DIRS} ${LIBCONFIG_INCLUDE_DIRS} ${ZLIB_INCLUDE_DIRS})

SET(LIBS ${LIBCONFIG_LIBRARIES} ${LIBEVENT_LIBRARIES} ${ZLIB_LIBRARIES} )

SET(SOURCES src/client.c src/config.c src/event_callback.c src/server.c src/log.c src/main.c src/portcheck.c src/util.c)

IF( ${USE_SQLITE3} )
    FIND_PACKAGE(Sqlite3 REQUIRED)
    LIST(APPEND INCLUDES ${SQLITE3_INCLUDE_DIRS})
    LIST(APPEND LIBS ${SQLITE3_LIBRARIES})
    LIST(APPEND SOURCES src/db_sqlite.c)
ELSEIF( ${USE_MYSQL} )
    MESSAGE(FATAL_ERROR "Not yet implemented")
ELSE()
    MESSAGE(FATAL_ERROR "Database backend not selected")
ENDIF()

INCLUDE_DIRECTORIES(${INCLUDES})
ADD_EXECUTABLE(ed2kd ${SOURCES})
TARGET_LINK_LIBRARIES(ed2kd ${LIBS})

INSTALL(TARGETS ed2kd
    RUNTIME DESTINATION bin
)
